public class 배열연습2 {

	public static void main(String[] args) {
		//산악동호회

		int[] ages = new int[10]; //나이 저장용

		ages[0] = 30;
		
		names[1] = "김길동";
		ages[1] = 35;

		System.out.println(names[0]);
		System.out.println(names[1]);
		System.out.println(names[2]);
	}

}
