```ruby

package day01;

public class 변수연습 {

	public static void main(String[] args) {
		int age = 37;
		String name = "Philip";
		System.out.println(name);
		System.out.println(age);
		
		System.out.println(name + "의 나이는 " + age + "세 입니다.");
	
		age = age + 1;
		System.out.println(name + "의 나이는 " + age + "세 입니다.");
	
		age = age - 5;
		System.out.println(name + "의 나이는 " + age + "세 입니다.");
	}

}
```
